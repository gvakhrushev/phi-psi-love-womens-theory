    address[] public members;
    mapping(address => uint256) public weights;  // œÜ –≤–µ—Å –∫–∞–∂–¥–æ–≥–æ
    
    address public phiDoors;  // –ì–ª–∞–≤–Ω—ã–π –ø—Ä–æ—Ç–æ–∫–æ–ª
    
    constructor(address[] memory _members, uint256[] memory _weights) {
        require(_members.length == _weights.length, "Length mismatch");
        
        members = _members;
        
        uint256 totalWeight = 0;
        for (uint256 i = 0; i < _members.length; i++) {
            weights[_members[i]] = _weights[i];
            totalWeight += _weights[i];
        }
        
        require(totalWeight == 1000, "Weights must sum to 100%");
        
        phiDoors = msg.sender;
    }
    
    /**
     * @notice –†–∞—Å–ø—Ä–µ–¥–µ–ª–∏—Ç—å –∑–∞—Ä–∞–±–æ—Ç–∞–Ω–Ω–æ–µ –º–µ–∂–¥—É —É—á–∞—Å—Ç–Ω–∏–∫–∞–º–∏
     */
    function distribute() external {
        uint256 balance = address(this).balance;
        
        for (uint256 i = 0; i < members.length; i++) {
            address member = members[i];
            uint256 share = (balance * weights[member]) / 1000;
            
            payable(member).transfer(share);
        }
        
        emit Distributed(balance);
    }
    
    /**
     * @notice –ì–æ–ª–æ—Å–æ–≤–∞—Ç—å –∫–∞–∫ –∫–æ–º–∞–Ω–¥–∞
     */

        (bool success,) = phiDoors.call{value: 0.03 ether}("");
        require(success, "Vote failed");
    }
    
    receive() external payable {}
    
    event Distributed(uint256 amount);
}
