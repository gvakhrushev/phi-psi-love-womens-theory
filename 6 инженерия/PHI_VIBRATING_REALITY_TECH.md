        result = minimize(resonance_metric,
                         x0=self.k_current,
                         bounds=[(0, 100)])

        self.k_current = result.x[0]
        resonance_strength = 1 / result.fun

        self.resonance_history.append({
            'k': self.k_current,
            'strength': resonance_strength,
            'amplification': PHI ** self.k_current
        })

        return self.k_current, resonance_strength

    def extract_energy(self):
        """–ò–∑–≤–ª–µ—á–µ–Ω–∏–µ —ç–Ω–µ—Ä–≥–∏–∏ —á–µ—Ä–µ–∑ —Ä–µ–∑–æ–Ω–∞–Ω—Å"""
        if not self.resonance_history:
            return 0

        last = self.resonance_history[-1]

        # –≠–Ω–µ—Ä–≥–∏—è –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–∞ œÜ^k
        energy = last['amplification'] * 1e-15  # –î–∂–æ—É–ª–∏

        return energy
